   list(APPEND EXTRA_LINGUIST_LDFLAGS '-Wl,-rpath,$ORIGIN/../lib')

   list(APPEND EXTRA_LINGUIST_LDFLAGS '-Wl,-rpath,$ORIGIN/../lib')

   PROPERTIES
   COMPILE_FLAGS ${EXTRA_LINGUIST_CXXFLAGS}

   EXPORT_NAME linguist
)
